#include "minishell.h"

void	print_open_fds(void)
{
	long	open_max = sysconf(_SC_OPEN_MAX);

	for(int i = 0; i < open_max; i++)
	{
		if (fcntl(i, F_GETFD) != -1 || errno!= EBADF)
			printf("File descriptor %d is open\n", i);
	}
}
